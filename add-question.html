function saveQuestion(){ try{ const q = document.getElementById("q"); const a1 = document.getElementById("a1"); const a2 = document.getElementById("a2"); const a3 = document.getElementById("a3"); const a4 = document.getElementById("a4"); const c = document.getElementById("c"); if(!q || !a1 || !a2 || !a3 || !a4 || !c){ alert("خطأ في عناصر الصفحة"); return; } if( !q.value.trim() || !a1.value.trim() || !a2.value.trim() || !a3.value.trim() || !a4.value.trim() ){ alert("أكمل جميع الحقول"); return; } const correct = parseInt(c.value); if(isNaN(correct) || correct < 0 || correct > 3){ alert("رقم الإجابة الصحيحة يجب أن يكون بين 0 و 3"); return; } if(editId){ let item = data.find(x => x.id === editId); if(!item) return; item.q = q.value.trim(); item.a = [a1.value, a2.value, a3.value, a4.value]; item.c = correct; editId = null; }else{ data.push({ id: nextId++, category: "غير محدد", level: "متوسط", q: q.value.trim(), a: [a1.value, a2.value, a3.value, a4.value], c: correct }); } clearForm(); render(); console.log("Question saved successfully"); }catch(err){ console.error(err); alert("حدث خطأ أثناء حفظ السؤال"); } }
